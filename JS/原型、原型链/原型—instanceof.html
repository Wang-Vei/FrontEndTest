instanceof

a instalceof b ———— a的原型链上是否有 b.prototype (new)

function A() { }
function B() { }
function C() { }

let c = new C() //c.__proto__ === C.prototype
B.prototype = c
let b = new B() //b.__proto__ === B.prototype
A.prototype = b
let a = new A() //a.__proto__ === A.prototype

console.log(a instanceof A) // true
console.log(a instanceof B) // true
console.log(a instanceof C) // true
console.log(a instanceof Object) //true

isPrototypeOf

b.isPrototypeOf(a) ———— a 的原型链上 是否有 b (setPrototypeOf)

let d = {}
let e = {}
let f = {}
Object.setPrototypeOf(e, f) //e.__proto__ === f
Object.setPrototypeOf(d, e) //d.__proto__ === e

console.log(e.isPrototypeOf(d)) //true
console.log(f.isPrototypeOf(e)) //true
console.log(f.isPrototypeOf(d)) //true
console.log(Object.prototype.isPrototypeOf(d)) // true
